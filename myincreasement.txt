from ontology.interop.System.Storage import GetContext,Get,Put
from ontology.interop.System.Runtime import Notify

ctx=GetContext()
INCREASEMENT_KEY='INCREASEMENT_KEY' 

def Main(operation,args):
    if operation == 'increase':
        return increase()
        
    return False
  
    
def increase():
    inc=Get(ctx,INCREASEMENT_KEY)
    if inc:
        inc=inc+1 
        
    else:
        inc=1
        
    
    Put(ctx,INCREASEMENT_KEY,inc)
    Notify(inc)
    return inc
